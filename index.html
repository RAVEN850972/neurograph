Простота использования

Создайте мощного ИИ-ассистента всего за несколько строк кода

Быстрый старт
Python
from neurograph.integration import create_default_engine

# Создание движка NeuroGraph
engine = create_default_engine()

# Обучение системы на новой информации
response = engine.learn( content="Python - мощный язык программирования для ИИ", source="user_input", importance=0.8 )

# Запрос к системе знаний
result = engine.query( question="Что такое Python и для чего он используется?", include_explanations=True )
print(result.primary_response)

# Output: "Python - мощный язык программирования, который широко # используется для разработки искусственного интеллекта..."

# Логический вывод с объяснениями
inference = engine.infer( premises=["Python используется для ИИ", "ИИ решает сложные задачи"], max_depth=3 )

# Корректное завершение работы
engine.shutdown()
